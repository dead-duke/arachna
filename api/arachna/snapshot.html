105    snapshot_files, to_files = _get_snapshot_files(snapshot_id, to_snapshot_id, root)
106    result = []
107    for s in sections:
108        if s.type in ("header",) or not s.path:
109            result.append(s)

111        lang = lang_for_path(Path(s.path))
112        parser = get_block_parser(lang)

118            _repo_map_deleted_section(s, lang, parser, snapshot_files, root)
119        result.append(s)
120    return result

34    fmt: OutputFormat = "markdown",

36    tokenizer=None,

def read_object(object_hash: str, root: pathlib.Path) -> bytes: 
67def read_object(object_hash: str, root: Path) -> bytes:
68    store_dir = _store_root(root)
69    path = _hash_path(store_dir, object_hash, mkdir=False)
70    if not path.exists():
71        raise ObjectNotFoundError(f"Object not found: {object_hash}")
72    raw = path.read_bytes()
73    try:
74        data = zlib.decompress(raw)
75    except zlib.error:
76        data = raw
77    actual_hash = hashlib.sha256(data).hexdigest()
78    if actual_hash != object_hash:
79        raise CorruptedStoreError(
80            f"Object {object_hash} is corrupted: expected hash {object_hash}, got {actual_hash}."
81        )
82    return data

241def rename_snapshot(old_id: str, new_id: str, root: Path) -> str:

255    manifest["name"] = new_id
256    atomic_write_text(new_path, json.dumps(manifest, indent=2) + "\n")
257    old_path.unlink()
258    head_path: SafePath = store_dir / "HEAD"
259    if head_path.exists() and head_path.read_text().strip() == old_id:
260        atomic_write_text(head_path, new_id + "\n")
