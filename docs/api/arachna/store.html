arachna.store

Content-addressable store for Watch snapshots.

Storage layout under .arachna/ (created lazily on first write):

.arachna/
  arachna.gitignore          # "*" — never commit this directory
  store/
    objects/
      ab/cd1234...    # zlib-compressed file content (if compression saves space)
    snapshots/
      20260603T103000.json   # manifest: {path: hash, ...}
    HEAD                     # text file with latest snapshot ID

Objects are stored by SHA256 hash. The hash is computed BEFORE optional zlib compression — so the hash identifies the content, not the encoding.

 29def _store_root() -> Path:
 30    """Return .arachna/store/ directory, creating it lazily.
 31
 32    On first creation, writes .arachna/.gitignore with "*" to prevent
 33    accidental commits of the store directory.
 34    """
 35    cwd = Path.cwd()
 36    arachna_dir = cwd / ".arachna"
 37    gitignore = arachna_dir / ".gitignore"
 38
 39    if not arachna_dir.is_dir():
 40        arachna_dir.mkdir(parents=True, exist_ok=True)
 41        gitignore.write_text("*\n")
 42
 43    store_dir = arachna_dir / "store"
 44    store_dir.mkdir(parents=True, exist_ok=True)
 45    return store_dir

 48def _hash_path(store_dir: Path, object_hash: str) -> Path:
 49    """Return path to object file: objects/ab/cd1234..."""
 50    objects_dir = store_dir / "objects"
 51    prefix = object_hash[:2]
 52    rest = object_hash[2:]
 53    obj_dir = objects_dir / prefix
 54    obj_dir.mkdir(parents=True, exist_ok=True)
 55    return obj_dir / rest

 58def write_object(data: bytes) -> str:
 59    """Store content in objects/, return SHA256 hash.
 60
 61    Hash is computed on original data BEFORE compression.
 62    If zlib compression reduces size, store compressed.
 63    Otherwise store raw.
 64    """
 65    object_hash = hashlib.sha256(data).hexdigest()
 66    store_dir = _store_root()
 67    path = _hash_path(store_dir, object_hash)
 68
 69    if path.exists():
 70        return object_hash
 71
 72    compressed = zlib.compress(data)
 73    if len(compressed) < len(data):
 74        path.write_bytes(compressed)
 75    else:
 76        path.write_bytes(data)
 77
 78    return object_hash

 81def read_object(object_hash: str) -> bytes:
 82    """Read from objects/, decompress if needed.
 83
 84    Auto-detects compression: tries zlib.decompress first,
 85    falls back to raw bytes if decompression fails (data stored uncompressed).
 86
 87    Raises:
 88        ObjectNotFoundError: object doesn't exist.
 89        CorruptedStoreError: content doesn't match hash.
 90    """
 91    store_dir = _store_root()
 92    path = _hash_path(store_dir, object_hash)
 93
 94    if not path.exists():
 95        raise ObjectNotFoundError(f"Object not found: {object_hash}")
 96
 97    raw = path.read_bytes()
 98
 99    # Auto-detect compression: try zlib, fallback to raw
100    try:
101        data = zlib.decompress(raw)
102    except zlib.error:
103        data = raw
104
105    # Verify integrity
106    actual_hash = hashlib.sha256(data).hexdigest()
107    if actual_hash != object_hash:
108        raise CorruptedStoreError(
109            f"Object {object_hash} is corrupted: expected {object_hash}, got {actual_hash}"
110        )
111
112    return data

115def create_snapshot(
116    files: dict[str, str],
117    profile_dict: dict | None = None,
118    profile: str = "full",
119    name: str | None = None,
120    pre_commands: dict[str, str] | None = None,
121    command: dict[str, str] | None = None,
122) -> str:
123    """Create a snapshot manifest and return its ID.
124
125    Args:
126        files: {path: content} dict of all files to snapshot.
127        profile_dict: full profile dict (directories, patterns, etc.).
128            Stored in manifest for later use by --diff without --profile.
129        profile: profile name string (legacy, used if profile_dict not given).
130        name: human-readable name. Required for named snapshots.
131            If None, timestamp YYYYMMDDTHHMMSS is used as ID.
132        pre_commands: optional {label: "sha256:hash"} for pre_commands output.
133        command: optional {label: "sha256:hash"} for command output.
134
135    Returns:
136        Snapshot ID (name if given, else timestamp YYYYMMDDTHHMMSS).
137
138    Raises:
139        SnapshotExistsError: if a snapshot with the given name already exists.
140    """
141    store_dir = _store_root()
142    snapshots_dir = store_dir / "snapshots"
143    snapshots_dir.mkdir(parents=True, exist_ok=True)
144
145    snapshot_id = name if name else datetime.now().strftime("%Y%m%dT%H%M%S")
146
147    # Check for duplicate name
148    manifest_path = snapshots_dir / f"{snapshot_id}.json"
149    if name and manifest_path.exists():
150        raise SnapshotExistsError(
151            f"Snapshot '{name}' already exists. Use 'arachna --snapshot update {name}' "
152            f"to update it, or delete it first."
153        )
154
155    # Store all file contents, collect hashes
156    file_hashes = {}
157    for path, content in files.items():
158        obj_hash = write_object(content.encode("utf-8"))
159        file_hashes[path] = f"sha256:{obj_hash}"
160
161    manifest = {
162        "id": snapshot_id,
163        "name": name,
164        "created": datetime.now().isoformat(),
165        "profile": profile_dict if profile_dict else profile,
166        "files": file_hashes,
167    }
168
169    if pre_commands:
170        manifest["pre_commands"] = pre_commands
171    if command:
172        manifest["command"] = command
173
174    manifest_path.write_text(json.dumps(manifest, indent=2))
175
176    # Update HEAD
177    head_path = store_dir / "HEAD"
178    head_path.write_text(snapshot_id)
179
180    return snapshot_id

183def update_snapshot(
184    snapshot_id: str,
185    files: dict[str, str],
186    profile_dict: dict | None = None,
187    pre_commands: dict[str, str] | None = None,
188    command: dict[str, str] | None = None,
189) -> str:
190    """Update an existing snapshot with new content.
191
192    Preserves snapshot ID and name. Updates created timestamp.
193    If profile_dict is given, updates stored profile.
194    Otherwise keeps existing profile from manifest.
195
196    Args:
197        snapshot_id: ID of the snapshot to update.
198        files: new {path: content} dict.
199        profile_dict: optional new profile dict.
200        pre_commands: optional new pre_commands data.
201        command: optional new command data.
202
203    Returns:
204        Snapshot ID (same as input).
205
206    Raises:
207        ObjectNotFoundError: if snapshot doesn't exist.
208    """
209    manifest = load_snapshot(snapshot_id)
210
211    # Store new file contents
212    file_hashes = {}
213    for path, content in files.items():
214        obj_hash = write_object(content.encode("utf-8"))
215        file_hashes[path] = f"sha256:{obj_hash}"
216
217    manifest["files"] = file_hashes
218    manifest["created"] = datetime.now().isoformat()
219
220    if profile_dict is not None:
221        manifest["profile"] = profile_dict
222
223    if pre_commands is not None:
224        manifest["pre_commands"] = pre_commands
225    elif "pre_commands" in manifest:
226        del manifest["pre_commands"]
227
228    if command is not None:
229        manifest["command"] = command
230    elif "command" in manifest:
231        del manifest["command"]
232
233    store_dir = _store_root()
234    manifest_path = store_dir / "snapshots" / f"{snapshot_id}.json"
235    manifest_path.write_text(json.dumps(manifest, indent=2))
236
237    # Update HEAD if this was HEAD
238    head_path = store_dir / "HEAD"
239    if head_path.exists() and head_path.read_text().strip() == snapshot_id:
240        head_path.write_text(snapshot_id)
241
242    return snapshot_id

262def list_snapshots() -> list[dict]:
263    """Return list of all manifest dicts, sorted by creation time descending."""
264    store_dir = _store_root()
265    snapshots_dir = store_dir / "snapshots"
266
267    if not snapshots_dir.is_dir():
268        return []
269
270    manifests = []
271    for mf in sorted(snapshots_dir.glob("*.json"), reverse=True):
272        try:
273            manifests.append(json.loads(mf.read_text()))
274        except (json.JSONDecodeError, OSError):
275            continue
276
277    manifests.sort(key=lambda m: m.get("created", ""), reverse=True)
278    return manifests

281def delete_snapshot(snapshot_id: str) -> None:
282    """Delete manifest file. Objects remain (cleaned by GC).
283
284    Raises:
285        ObjectNotFoundError: snapshot doesn't exist.
286    """
287    store_dir = _store_root()
288    manifest_path = store_dir / "snapshots" / f"{snapshot_id}.json"
289
290    if not manifest_path.exists():
291        raise ObjectNotFoundError(f"Snapshot not found: {snapshot_id}")
292
293    manifest_path.unlink()
294
295    # Clear HEAD if it pointed to this snapshot
296    head_path = store_dir / "HEAD"
297    if head_path.exists() and head_path.read_text().strip() == snapshot_id:
298        # Point HEAD to the latest remaining snapshot, or remove
299        remaining = list_snapshots()
300        if remaining:
301            head_path.write_text(remaining[0]["id"])
302        else:
303            head_path.unlink()

306def rename_snapshot(old_id: str, new_id: str) -> str:
307    """Rename a snapshot — updates manifest id/name and file name.
308
309    Updates HEAD if it pointed to the old snapshot ID.
310
311    Args:
312        old_id: current snapshot ID.
313        new_id: new snapshot ID.
314
315    Returns:
316        New snapshot ID.
317
318    Raises:
319        ObjectNotFoundError: if old_id doesn't exist.
320        SnapshotExistsError: if new_id already exists.
321    """
322    store_dir = _store_root()
323    snapshots_dir = store_dir / "snapshots"
324    old_path = snapshots_dir / f"{old_id}.json"
325    new_path = snapshots_dir / f"{new_id}.json"
326
327    if not old_path.exists():
328        raise ObjectNotFoundError(f"Snapshot not found: {old_id}")
329
330    if new_path.exists():
331        raise SnapshotExistsError(
332            f"Snapshot '{new_id}' already exists. "
333            f"Use a different name or delete the existing snapshot first."
334        )
335
336    # Load manifest, update id and name
337    manifest = json.loads(old_path.read_text())
338    manifest["id"] = new_id
339    manifest["name"] = new_id
340
341    # Write to new path
342    new_path.write_text(json.dumps(manifest, indent=2))
343
344    # Remove old file
345    old_path.unlink()
346
347    # Update HEAD if it pointed to old_id
348    head_path = store_dir / "HEAD"
349    if head_path.exists() and head_path.read_text().strip() == old_id:
350        head_path.write_text(new_id)
351
352    return new_id

355def gc() -> dict:
356    """Garbage collection: delete unreferenced objects.
357
358    Finds all hashes referenced across all manifests,
359    deletes object files not in the referenced set.
360
361    Returns:
362        {removed: N, freed_bytes: M}
363    """
364    store_dir = _store_root()
365    objects_dir = store_dir / "objects"
366    snapshots_dir = store_dir / "snapshots"
367
368    if not objects_dir.is_dir():
369        return {"removed": 0, "freed_bytes": 0}
370
371    # Collect all referenced hashes
372    referenced = set()
373    if snapshots_dir.is_dir():
374        for mf in snapshots_dir.glob("*.json"):
375            try:
376                manifest = json.loads(mf.read_text())
377                for hash_spec in manifest.get("files", {}).values():
378                    # Format: "sha256:abcdef..."
379                    if hash_spec.startswith("sha256:"):
380                        referenced.add(hash_spec[7:])
381                for hash_spec in manifest.get("pre_commands", {}).values():
382                    if hash_spec.startswith("sha256:"):
383                        referenced.add(hash_spec[7:])
384                for hash_spec in manifest.get("command", {}).values():
385                    if hash_spec.startswith("sha256:"):
386                        referenced.add(hash_spec[7:])
387            except (json.JSONDecodeError, OSError):
388                continue
389
390    # Delete unreferenced objects
391    removed = 0
392    freed_bytes = 0
393    for obj_file in objects_dir.rglob("*"):
394        if not obj_file.is_file():
395            continue
396        # Object path: objects/ab/cd1234... → hash = ab + cd1234...
397        rel = obj_file.relative_to(objects_dir)
398        obj_hash = str(rel).replace(os.sep, "")
399        if obj_hash not in referenced:
400            with contextlib.suppress(OSError):
401                freed_bytes += obj_file.stat().st_size
402            obj_file.unlink()
403            removed += 1
404
405    return {"removed": removed, "freed_bytes": freed_bytes}

408def stats() -> dict:
409    """Return store statistics.
410
411    Returns:
412        {snapshots: N, objects: M, total_bytes: X, unique_bytes: Y, dedup_pct: Z}
413    """
414    store_dir = _store_root()
415    objects_dir = store_dir / "objects"
416    snapshots_dir = store_dir / "snapshots"
417
418    snapshots_count = len(list(snapshots_dir.glob("*.json"))) if snapshots_dir.is_dir() else 0
419
420    objects_count = 0
421    total_bytes = 0
422    if objects_dir.is_dir():
423        for obj_file in objects_dir.rglob("*"):
424            if obj_file.is_file():
425                objects_count += 1
426                with contextlib.suppress(OSError):
427                    total_bytes += obj_file.stat().st_size
428
429    # Unique bytes: sum of sizes of objects referenced by any snapshot
430    referenced_hashes = set()
431    if snapshots_dir.is_dir():
432        for mf in snapshots_dir.glob("*.json"):
433            try:
434                manifest = json.loads(mf.read_text())
435                for hash_spec in manifest.get("files", {}).values():
436                    if hash_spec.startswith("sha256:"):
437                        referenced_hashes.add(hash_spec[7:])
438                for hash_spec in manifest.get("pre_commands", {}).values():
439                    if hash_spec.startswith("sha256:"):
440                        referenced_hashes.add(hash_spec[7:])
441                for hash_spec in manifest.get("command", {}).values():
442                    if hash_spec.startswith("sha256:"):
443                        referenced_hashes.add(hash_spec[7:])
444            except (json.JSONDecodeError, OSError):
445                continue
446
447    unique_bytes = 0
448    for obj_hash in referenced_hashes:
449        obj_path = _hash_path(store_dir, obj_hash)
450        if obj_path.exists():
451            with contextlib.suppress(OSError):
452                unique_bytes += obj_path.stat().st_size
453
454    dedup_pct = 0.0
455    if total_bytes > 0:
456        dedup_pct = round((1 - unique_bytes / total_bytes) * 100, 1)
457
458    return {
459        "snapshots": snapshots_count,
460        "objects": objects_count,
461        "total_bytes": total_bytes,
462        "unique_bytes": unique_bytes,
463        "dedup_pct": dedup_pct,
464    }

307def rename_snapshot(old_id: str, new_id: str) -> str:
308    """Rename a snapshot — updates manifest id/name and file name.
309
310    Updates HEAD if it pointed to the old snapshot ID.
311
312    Args:
313        old_id: current snapshot ID.
314        new_id: new snapshot ID.
315
316    Returns:
317        New snapshot ID.
318
319    Raises:
320        ObjectNotFoundError: if old_id doesn't exist.
321        SnapshotExistsError: if new_id already exists.
322    """
323    store_dir = _store_root()
324    snapshots_dir = store_dir / "snapshots"
325    old_path = snapshots_dir / f"{old_id}.json"
326    new_path = snapshots_dir / f"{new_id}.json"
327
328    if not old_path.exists():
329        raise ObjectNotFoundError(f"Snapshot not found: {old_id}")
330
331    if new_path.exists():
332        raise SnapshotExistsError(
333            f"Snapshot '{new_id}' already exists. "
334            f"Use a different name or delete the existing snapshot first."
335        )
336
337    # Load manifest, update id and name
338    manifest = json.loads(old_path.read_text())
339    manifest["id"] = new_id
340    manifest["name"] = new_id
341
342    # Write to new path
343    new_path.write_text(json.dumps(manifest, indent=2))
344
345    # Remove old file
346    old_path.unlink()
347
348    # Update HEAD if it pointed to old_id
349    head_path = store_dir / "HEAD"
350    if head_path.exists() and head_path.read_text().strip() == old_id:
351        head_path.write_text(new_id)
352
353    return new_id

def gc() -> dict: 
356def gc() -> dict:
357    """Garbage collection: delete unreferenced objects.
358
359    Finds all hashes referenced across all manifests,
360    deletes object files not in the referenced set.
361
362    Returns:
363        {removed: N, freed_bytes: M}
364    """
365    store_dir = _store_root()
366    objects_dir = store_dir / "objects"
367    snapshots_dir = store_dir / "snapshots"
368
369    if not objects_dir.is_dir():
370        return {"removed": 0, "freed_bytes": 0}
371
372    # Collect all referenced hashes
373    referenced = set()
374    if snapshots_dir.is_dir():
375        for mf in snapshots_dir.glob("*.json"):
376            try:
377                manifest = json.loads(mf.read_text())
378                for hash_spec in manifest.get("files", {}).values():
379                    # Format: "sha256:abcdef..."
380                    if hash_spec.startswith("sha256:"):
381                        referenced.add(hash_spec[7:])
382                for hash_spec in manifest.get("pre_commands", {}).values():
383                    if hash_spec.startswith("sha256:"):
384                        referenced.add(hash_spec[7:])
385                for hash_spec in manifest.get("command", {}).values():
386                    if hash_spec.startswith("sha256:"):
387                        referenced.add(hash_spec[7:])
388            except (json.JSONDecodeError, OSError):
389                continue
390
391    # Delete unreferenced objects
392    removed = 0
393    freed_bytes = 0
394    for obj_file in objects_dir.rglob("*"):
395        if not obj_file.is_file():
396            continue
397        # Object path: objects/ab/cd1234... → hash = ab + cd1234...
398        rel = obj_file.relative_to(objects_dir)
399        obj_hash = str(rel).replace(os.sep, "")
400        if obj_hash not in referenced:
401            with contextlib.suppress(OSError):
402                freed_bytes += obj_file.stat().st_size
403            obj_file.unlink()
404            removed += 1
405
406    return {"removed": removed, "freed_bytes": freed_bytes}
