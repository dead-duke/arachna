def _path_matches_profile(path, profile, root): 
557def _path_matches_profile(path, profile, root):
558    """Check if a file path belongs to the given profile's directories/patterns/files."""
559    normalized_files = [_rel_path(Path(f), root) for f in profile.get("files", [])]
560    if path in normalized_files:
561        return True
562    directories = profile.get("directories", [])
563    patterns = profile.get("patterns", ["*"])
564    path_obj = Path(path)
565    for directory in directories:
566        dir_path = Path(directory)
567        try:
568            path_obj.relative_to(dir_path)
569        except ValueError:
570            continue
571        for pat in patterns:
572            if fnmatch.fnmatch(path_obj.name, pat):
573                return True
574    return False
